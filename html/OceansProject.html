# STEP 1: Imports
import earthaccess
import xarray as xr
import matplotlib.pyplot as plt
import numpy as np
import pandas as pd
import seaborn as sns

sns.set(style="whitegrid")
plt.rcParams["figure.figsize"] = (10, 5)

# STEP 2: Login
print(" Logging into NASA Earthdata...")
earthaccess.login(strategy="interactive", persist=True)

# STEP 3: Define Filters
temporal_range = ("2023-07-01T00:00:00Z", "2023-07-01T01:00:00Z")  # 1-hour window
bounding_box = (-5.0, 85.0, 5.0, 89.0)  # Valid region (lat, lon)

# STEP 4: Search MODIS SST
print(" Searching MODIS SST...")
modis_results = earthaccess.search_data(
    short_name="MODIS_A-JPL-L2P-v2019.0",
    cloud_hosted=True,
    temporal=temporal_range,
    bounding_box=bounding_box
)

modis_results = modis_results[:1]  # Limit granules
print(f" MODIS granules: {len(modis_results)}")

# STEP 5: Load dataset via streaming
modis_ds = xr.open_dataset(earthaccess.open(modis_results)[0])

# STEP 6: Extract SST data
sst = modis_ds["sea_surface_temperature"][0]
sst = sst.where(sst < 1000)  # Remove fill values

# STEP 7: Auto-detect coordinate names
def get_coord_names(da):
    coords = list(da.coords)
    lat = [c for c in coords if 'lat' in c.lower()][0]
    lon = [c for c in coords if 'lon' in c.lower()][0]
    return lat, lon

lat_name, lon_name = get_coord_names(sst)
lat = sst[lat_name]
lon = sst[lon_name]

# STEP 8: Subset and downsample
mask = (lat > -2) & (lat < 2)
lat_idx = np.where(mask)[0]
mask = (lon > 85) & (lon < 89)
lon_idx = np.where(mask)[0]

if len(lat_idx) > 0 and len(lon_idx) > 0:
    sst_subset = sst.isel(
        {lat_name: lat_idx[::2], lon_name: lon_idx[::2]}  # Downsample every 2 points
    )
else:
    print(" Subsetting skipped: no valid region found.")
    sst_subset = sst

# STEP 9: Plot
plt.figure()
sst_subset.plot(cmap="coolwarm")
plt.title(" MODIS Sea Surface Temperature (Subset)")
plt.show()

# STEP 10: Stats (with safe check)
def print_stats(name, da):
    vals = da.values.flatten()
    vals = vals[np.isfinite(vals)]
    print(f"\n {name} Stats:")
    if len(vals) == 0:
        print(" No valid data in selected region.")
    else:
        print(f" Min: {np.min(vals):.2f}")
        print(f" Max: {np.max(vals):.2f}")
        print(f" Mean: {np.mean(vals):.2f}")

print_stats("MODIS SST", sst_subset)

# STEP 11: Histogram
vals = sst_subset.values.flatten()
vals = vals[np.isfinite(vals)]

if len(vals) > 0:
    plt.figure()
    sns.histplot(vals, bins=50, kde=True)
    plt.title("MODIS SST Histogram")
    plt.xlabel("Temperature (K)")
    plt.grid(True)
    plt.show()
else:
    print(" No data to plot histogram.")

print("\n Script finished with safe bounding box and data filtering.")
